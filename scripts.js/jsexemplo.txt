const formDados = document.querySelector('#formulario') // tá pegando o id do formulario
const divResultado = document.querySelector('#resultado')
//pegou id do div resultado

//eventos em uma página - comportamentos do usuario - click, clique duplo, submeter informações, etc

//

formDados.addEventListener('submit', (evt) => {
    evt.preventDefault()

    const form_num = new FormData(formDados)

    let nome = (form_num.get('nome'))
    let num1 = parseFloat(form_num.get('num1'))
    let num2 = parseFloat(form_num.get('num2'))


    let IMC = (num1 / (num2 * num2))

    let resultIMC = ""

    if (IMC < 20) {
        resultIMC = "ABAIXO DO PESO"
    }
    else if (IMC >= 20 && IMC < 25) {
        resultIMC = "NORMAL"
    }
    else if (IMC >= 25 && IMC < 30) {
        resultIMC = "EXCESSO DE PESO"
    }
    else if (IMC >= 30 && IMC < 35) {
        resultIMC = "OBESIDADE"
    }
    else {
        resultIMC = "OBESIDADE MÓRBIDA"
    }

    IMCn = IMC.toFixed(1);

    divResultado.innerHTML = `${nome}, seu IMC é ${IMCn}, logo, sua situação é: ${resultIMC}`
})
